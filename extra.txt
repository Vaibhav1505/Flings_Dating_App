    final body = res.body;
    final jsonData = jsonDecode(body);
    final results = jsonData['results'] as List<dynamic>;

    setState(() {
      users = results.map((e) {
        final username = Username(

        return User(
            email: e['email'],
            gender: e['gender'],
            phone: e['phone'],
            nat: e["nat"],
            name: username);
      }).toList();
    });
